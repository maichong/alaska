# alaska-field-image
Alaska image field

```javascript
//alaska config

export default {

  'alaska-field-image': {
    allowed: ['png','jpg','bmp'],
    multi: false,
    //empty or ends with /
    dir: 'path/to/',
    //empty or ends with /
    pathFormat: 'YYYY/MM/DD/',
    //empty or ends with /
    prefix: 'http://maichong.it/'
  }

}
```
