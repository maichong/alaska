# alaska-field-text
Alaska text field

```javascript
//alaska config

export default {
  'alaska-field-text': {
    //是否全部转为小写
    lowercase: false,
    //是否全部转为大写
    uppercase: false,
    //是否去掉首尾空格
    trim: false,
    //字符串模式
    match: /^[a-z]\w*/i,
    //字段最大长度
    maxlength: 20,
    //字段最小长度
    minlength: 1
  }
}
```
